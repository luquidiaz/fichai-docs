---
title: 'Listar Desarrollos'
api: 'GET /desarrollos'
description: 'Obtiene una lista paginada de desarrollos inmobiliarios'
---

## Descripción

Retorna una lista paginada de desarrollos inmobiliarios públicos listados en el MLS de FichAI. Solo incluye desarrollos con `mls_fichai = true`.

## Parámetros

<ParamField query="country" type="string">
  Código de país ISO 3166-1 alpha-2 para filtrar.

  Ejemplos: `AR` (Argentina), `UY` (Uruguay), `MX` (México)
</ParamField>

<ParamField query="org" type="string">
  Slug de la organización para filtrar. Solo retorna desarrollos de esa inmobiliaria.
</ParamField>

<ParamField query="status" type="string" default="disponible,en_construccion">
  Estados a incluir, separados por coma.

  Valores posibles: `disponible`, `en_construccion`, `vendido`, `pausado`
</ParamField>

<ParamField query="limit" type="integer" default="50">
  Cantidad de resultados por página. Mínimo: 1, Máximo: 100.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Número de resultados a saltar para paginación.
</ParamField>

## Respuesta

<ResponseField name="desarrollos" type="array">
  Array de objetos desarrollo.

  <Expandable title="Propiedades de cada desarrollo">
    <ResponseField name="id" type="string">
      UUID único del desarrollo
    </ResponseField>
    <ResponseField name="titulo" type="string">
      Nombre del desarrollo
    </ResponseField>
    <ResponseField name="slug" type="string">
      Identificador URL-friendly
    </ResponseField>
    <ResponseField name="descripcion" type="string">
      Descripción del desarrollo
    </ResponseField>
    <ResponseField name="estado" type="string">
      Estado actual: `disponible`, `en_construccion`, `vendido`, `pausado`
    </ResponseField>
    <ResponseField name="etapa_desarrollo" type="string">
      Etapa de construcción: `pozo`, `en_construccion`, `terminado`
    </ResponseField>
    <ResponseField name="fecha_entrega" type="string">
      Fecha estimada de entrega (ISO 8601)
    </ResponseField>
    <ResponseField name="country_code" type="string">
      Código de país ISO
    </ResponseField>
    <ResponseField name="localizacion" type="object">
      Ubicación del desarrollo
    </ResponseField>
    <ResponseField name="multimedia" type="object">
      Imágenes, videos y tours 360
    </ResponseField>
    <ResponseField name="precios" type="object">
      Rango de precios y moneda
    </ResponseField>
    <ResponseField name="organizations" type="object">
      Información de la inmobiliaria
    </ResponseField>
    <ResponseField name="created_at" type="string">
      Fecha de creación (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="meta" type="object">
  Metadata de paginación

  <Expandable title="Propiedades">
    <ResponseField name="total" type="integer">
      Total de resultados disponibles
    </ResponseField>
    <ResponseField name="limit" type="integer">
      Límite usado en esta request
    </ResponseField>
    <ResponseField name="offset" type="integer">
      Offset usado en esta request
    </ResponseField>
    <ResponseField name="has_more" type="boolean">
      `true` si hay más resultados
    </ResponseField>
  </Expandable>
</ResponseField>

```python Python
)

data = response.json()
```
