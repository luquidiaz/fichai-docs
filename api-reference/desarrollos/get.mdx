---
title: 'Obtener Desarrollo'
api: 'GET /desarrollos/{slug}'
description: 'Obtiene el detalle completo de un desarrollo con sus unidades'
---

## Descripción

Retorna información detallada de un desarrollo específico, incluyendo sus unidades disponibles y estadísticas de venta. Solo funciona para desarrollos con `mls_fichai = true`.

## Parámetros

<ParamField path="slug" type="string" required>
  Slug único del desarrollo. Es el identificador URL-friendly.

  Ejemplo: `torre-del-parque`
</ParamField>

## Respuesta

<ResponseField name="desarrollo" type="object">
  Objeto con toda la información del desarrollo.

  <Expandable title="Propiedades">
    <ResponseField name="id" type="string">
      UUID único del desarrollo
    </ResponseField>
    <ResponseField name="titulo" type="string">
      Nombre del desarrollo
    </ResponseField>
    <ResponseField name="slug" type="string">
      Identificador URL-friendly
    </ResponseField>
    <ResponseField name="descripcion" type="string">
      Descripción completa del desarrollo
    </ResponseField>
    <ResponseField name="estado" type="string">
      Estado actual
    </ResponseField>
    <ResponseField name="etapa_desarrollo" type="string">
      Etapa de construcción
    </ResponseField>
    <ResponseField name="fecha_entrega" type="string">
      Fecha estimada de entrega
    </ResponseField>
    <ResponseField name="country_code" type="string">
      Código de país
    </ResponseField>
    <ResponseField name="localizacion" type="object">
      Ubicación completa
    </ResponseField>
    <ResponseField name="multimedia" type="object">
      Imágenes, videos y tours
    </ResponseField>
    <ResponseField name="precios" type="object">
      Rango de precios
    </ResponseField>
    <ResponseField name="organizations" type="object">
      Información de la inmobiliaria
    </ResponseField>
    <ResponseField name="unidades" type="array">
      Array de unidades disponibles/reservadas

      <Expandable title="Propiedades de cada unidad">
        <ResponseField name="id" type="string">
          UUID de la unidad
        </ResponseField>
        <ResponseField name="titulo" type="string">
          Nombre/identificador de la unidad
        </ResponseField>
        <ResponseField name="estado" type="string">
          `disponible`, `reservada`, o `vendida`
        </ResponseField>
        <ResponseField name="precios" type="object">
          Precio de la unidad
        </ResponseField>
        <ResponseField name="multimedia" type="object">
          Imágenes de la unidad
        </ResponseField>
        <ResponseField name="piso" type="integer">
          Número de piso
        </ResponseField>
        <ResponseField name="numero" type="string">
          Número/letra de unidad
        </ResponseField>
        <ResponseField name="orientacion" type="string">
          Orientación (Norte, Sur, etc.)
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="unidades_stats" type="object">
      Estadísticas de unidades

      <Expandable title="Propiedades">
        <ResponseField name="total_unidades" type="integer">
          Total de unidades en el desarrollo
        </ResponseField>
        <ResponseField name="disponibles" type="integer">
          Unidades disponibles
        </ResponseField>
        <ResponseField name="reservadas" type="integer">
          Unidades reservadas
        </ResponseField>
        <ResponseField name="vendidas" type="integer">
          Unidades vendidas
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://app.fichai.com/api/public/desarrollos/torre-del-parque" \
  -H "Authorization: Bearer pk_live_xxxxxxxxxxxx"
```

```javascript JavaScript
const response = await fetch(
  'https://app.fichai.com/api/public/desarrollos/torre-del-parque',
  {
    headers: {
      'Authorization': 'Bearer pk_live_xxxxxxxxxxxx'
    }
  }
);

const { desarrollo } = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://app.fichai.com/api/public/desarrollos/torre-del-parque',
    headers={'Authorization': 'Bearer pk_live_xxxxxxxxxxxx'}
)

desarrollo = response.json()['desarrollo']
```
</RequestExample>
